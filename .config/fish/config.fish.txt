if status is-interactive
    # No greeting
    set fish_greeting

    # Use starship
    function starship_transient_prompt_func
        starship module character
    end
    if test "$TERM" != linux
        starship init fish | source
        enable_transience
    end

    # Colors
    if test -f ~/.local/state/quickshell/user/generated/terminal/sequences.txt
        cat ~/.local/state/quickshell/user/generated/terminal/sequences.txt
    end

    # Aliases
    alias clear "printf '\033[2J\033[3J\033[1;1H'"
    alias celar "printf '\033[2J\033[3J\033[1;1H'"
    alias claer "printf '\033[2J\033[3J\033[1;1H'"
    alias pamcan pacman
    alias q 'qs -c ii'
    # alias sudo="sudo-rs"

    if test "$TERM" != linux
        if type -q eza
            alias ls='eza --icons --group-directories-first'
            alias ll='eza -l --icons --group-directories-first'
            alias la='eza -a --icons --group-directories-first'
        else
            alias ls 'ls --color=auto'
        end
    end

    if test "$TERM" = xterm-kitty
        alias ssh 'kitten ssh'
    end

    # Initialisierungen
    zoxide init fish | source
    thefuck --alias | source

set -gx EDITOR nvim

# Yazi Wrapper-Funktion (ändert das Verzeichnis beim Beenden)
function y
    set tmp (mktemp -t "yazi-cwd.XXXXXX")
    yazi $argv --cwd-file="$tmp"
    if set cwd (command cat -- "$tmp"); and [ -n "$cwd" ]; and [ "$cwd" != "$PWD" ]
        builtin cd -- "$cwd"
    end
    rm -f -- "$tmp"
end

# Keybindings
function fish_user_key_bindings
    # Vi Mode aktivieren
    fish_vi_key_bindings

    # Autopair Plugin falls vorhanden
    if functions -q _autopair_install
        _autopair_install
    end

    # Standard fzf Bindings laden
    fzf_key_bindings

    # 1. Bestehende Bindings für Alt+L löschen
    bind -e \el
    bind -M insert -e \el
    bind -M default -e \el

    # 2. Wort-Navigation (Alt+H, Alt+L)
    bind -M insert \eh prevd-or-backward-word
    bind -M insert \el nextd-or-forward-word
    bind -M default \eh prevd-or-backward-word
    bind -M default \el nextd-or-forward-word

    # 3. Schnelle Befehle (Strg+S für ls)
    bind -M insert \cs "commandline -r 'ls -laa'; commandline -f execute"
    bind -M default \cs "commandline -r 'ls -laa'; commandline -f execute"

    # 4. App-Shortcuts (Strg+F, Strg+N, Strg+G)
    bind -M insert \cf 'yazi; commandline -f repaint'
    bind -M default \cf 'yazi; commandline -f repaint'
    
    bind -M insert \cn 'nvim (pwd); commandline -f repaint'
    bind -M default \cn 'nvim (pwd); commandline -f repaint'

    bind -M insert \cg 'lazygit; commandline -f repaint'
    bind -M default \cg 'lazygit; commandline -f repaint'
end
